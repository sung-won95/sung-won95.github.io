docker-compose에서 NFS 사용하기

  • Docker-compose Setting

Docker-compose Setting

version: '3.1'

volumes:
  nfs-data:                 #nfs volume 이름 지정 
    driver: local   
    driver_opts:            
      type: nfs             #nfs 서비스로 지정 
      device: ":/your/nfs/path"     #nfs mount 경로
      o: "addr=192.168.2.27,nolock,soft,rw" #nfs서버 세팅

services:
  AI-blur:
    image: blur
    deploy:
      resources:
        limits:
          cpus: '24'
          memory: 128G
    build:
      context: .
      dockerfile: blur/Dockerfile
    environment:
      - NVIDIA_VISIBLE_DEVICES=2,3,4,5    #GPU 설정 env
    volumes:       
      - nfs-blur:/root/workspace/nfs-data  # [volume명]:[컨테이너 내부 마운트 위치]
    stdin_open: true
    tty: true

docker-compose에서도 위와같이 nfs를 사용 할 수 있습니다!

추가적으로 궁금하신건 아래 링크 중 volume 부분 참고하시면 됩니다
